🧠 MiniBrain MemoryService Architecture

Executive Summary

The MemoryService is a critical component of MiniBrain that provides persistent memory storage and retrieval capabilities using Qdrant vector database. This service enables MiniBrain to remember all previous conversations, maintain context across sessions, and provide intelligent retrieval of relevant past interactions using a custom local embedding service.

Key Features:
  • Vector-based semantic search for conversation retrieval
  • Hierarchical memory organization (Session → Conversation → Message)
  • Real-time memory updates during conversations
  • Context-aware memory retrieval with intelligent chunking
  • Efficient similarity search using local custom embeddings

Implementation Details

Memory Service Implementation

public class MemoryService : IMemoryService
{
    private readonly IQdrantClient _qdrantClient;
    private readonly IEmbeddingService _embeddingService;
    private readonly ILogger<MemoryService> _logger;
    private const string COLLECTION_NAME = "minibrain_memories";
    
    public async Task<string> StoreMemoryAsync(Memory memory)
    {
        memory.Id = Guid.NewGuid().ToString();
        memory.Embedding = await _embeddingService.GenerateEmbeddingAsync(memory.Content);
        
        var point = new PointStruct
        {
            Id = new PointId { Uuid = memory.Id },
            Vectors = memory.Embedding,
            Payload = new Dictionary<string, object>
            {
                ["conversation_id"] = memory.ConversationId,
                ["session_id"] = memory.SessionId,
                ["content"] = memory.Content,
                ["role"] = memory.Role,
                ["timestamp"] = memory.Timestamp,
                ["metadata"] = JsonSerializer.Serialize(memory.Metadata)
            }
        };
        
        await _qdrantClient.UpsertAsync(COLLECTION_NAME, new[] { point });
        return memory.Id;
    }
}

Custom Embedding Service Configuration

public class CustomEmbeddingService : IEmbeddingService
{
    private readonly ILogger<CustomEmbeddingService> _logger;
    private readonly ISemanticChunker _semanticChunker;
    private readonly ConcurrentDictionary<string, float[]> _embeddingCache;
    private readonly int _embeddingDimension; // Loaded from Qdrant.VectorSize config
    
    public CustomEmbeddingService(ILogger<CustomEmbeddingService> logger, 
                                 ISemanticChunker semanticChunker, 
                                 IConfiguration configuration)
    {
        _logger = logger;
        _semanticChunker = semanticChunker;
        _embeddingCache = new ConcurrentDictionary<string, float[]>();
        
        // Load embedding dimension from same config as Qdrant
        var qdrantConfig = configuration.GetSection("Qdrant");
        _embeddingDimension = int.Parse(qdrantConfig["VectorSize"] ?? "512");
    }
    
    public async Task<float[]> GenerateEmbeddingAsync(string text)
    {
        if (string.IsNullOrWhiteSpace(text))
            return new float[_embeddingDimension];

        var normalizedText = NormalizeText(text);
        var cacheKey = GenerateCacheKey(normalizedText);
        
        if (_embeddingCache.TryGetValue(cacheKey, out var cachedEmbedding))
            return cachedEmbedding;

        var embedding = await ComputeEmbeddingAsync(normalizedText);
        _embeddingCache.TryAdd(cacheKey, embedding);
        return embedding;
    }
    
    public async Task<List<TextChunk>> ChunkTextAsync(string text, int maxChunkSize = 1000, int overlapSize = 200)
    {
        return await _semanticChunker.ChunkTextAsync(text, maxChunkSize, overlapSize);
    }
}

Strategy Implementation

public async Task<List<Memory>> SearchSimilarMemoriesAsync(string content, float threshold = 0.8f)
{
    var queryEmbedding = await _embeddingService.GenerateEmbeddingAsync(content);
    
    var searchResult = await _qdrantClient.SearchAsync(new SearchRequest
    {
        Vector = queryEmbedding,
        Limit = 20,
        ScoreThreshold = threshold,
        WithPayload = true,
        Filter = new Filter
        {
            Must = new[]
            {
                new FieldCondition
                {
                    Key = "timestamp",
                    Range = new RangeCondition
                    {
                        Gte = DateTime.UtcNow.AddDays(-30)
                    }
                }
            }
        }
    });
    
    return searchResult.Select(ConvertToMemory).ToList();
}

Security & Privacy

Critical Security Measures:
  • All memories encrypted at rest in Qdrant
  • TLS encryption for all network communication
  • API key rotation every 90 days
  • Audit logging for all memory operations
  • PII detection and masking before storage

Conclusion

The MemoryService architecture provides a robust, scalable solution for persistent conversation memory in MiniBrain. By leveraging Qdrant's vector search capabilities and implementing a custom local embedding service, the system can efficiently store and retrieve relevant memories to enhance the AI's contextual understanding.

Key Takeaways:
  • Local custom embedding service eliminates external API dependencies
  • 512-dimension vectors provide efficient semantic similarity search
  • Hierarchical organization maintains conversation context
  • Chunking and caching optimize performance for large content
  • Caching and optimization ensure low-latency responses
  • Security measures protect sensitive conversation data
  • Extensible architecture supports future enhancements
