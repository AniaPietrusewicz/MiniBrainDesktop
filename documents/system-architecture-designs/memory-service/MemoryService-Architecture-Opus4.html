🧠 MiniBrain MemoryService Architecture

Executive Summary

The MemoryService is a critical component of MiniBrain that provides persistent memory storage and retrieval capabilities using Qdrant vector database. This service enables MiniBrain to remember all previous conversations, maintain context across sessions, and provide intelligent retrieval of relevant past interactions.

Key Features:
  • Vector-based semantic search for conversation retrieval
  • Hierarchical memory organization (Session → Conversation → Message)
  • Real-time memory updates during conversations
  • Context-aware memory retrieval
  • Efficient similarity search using embeddings

Implementation Details

Memory Service Implementation

public class MemoryService : IMemoryService
{
    private readonly IQdrantClient _qdrantClient;
    private readonly IEmbeddingService _embeddingService;
    private readonly ILogger<MemoryService> _logger;
    private const string COLLECTION_NAME = "minibrain_memories";
    
    public async Task<string> StoreMemoryAsync(Memory memory)
    {
        memory.Id = Guid.NewGuid().ToString();
        memory.Embedding = await _embeddingService.GenerateEmbeddingAsync(memory.Content);
        
        var point = new PointStruct
        {
            Id = new PointId { Uuid = memory.Id },
            Vectors = memory.Embedding,
            Payload = new Dictionary<string, object>
            {
                ["conversation_id"] = memory.ConversationId,
                ["session_id"] = memory.SessionId,
                ["content"] = memory.Content,
                ["role"] = memory.Role,
                ["timestamp"] = memory.Timestamp,
                ["metadata"] = JsonSerializer.Serialize(memory.Metadata)
            }
        };
        
        await _qdrantClient.UpsertAsync(COLLECTION_NAME, new[] { point });
        return memory.Id;
    }
}

Embedding Service Configuration

public class EmbeddingService : IEmbeddingService
{
    private readonly OpenAIClient _openAIClient;
    private const string MODEL = "text-embedding-ada-002";
    
    public async Task<float[]> GenerateEmbeddingAsync(string text)
    {
        var response = await _openAIClient.Embeddings.CreateAsync(new EmbeddingCreateRequest
        {
            Model = MODEL,
            Input = text
        });
        
        return response.Data[0].Embedding.ToArray();
    }
}

Strategy Implementation

public async Task<List<Memory>> SearchSimilarMemoriesAsync(string content, float threshold = 0.8f)
{
    var queryEmbedding = await _embeddingService.GenerateEmbeddingAsync(content);
    
    var searchResult = await _qdrantClient.SearchAsync(new SearchRequest
    {
        Vector = queryEmbedding,
        Limit = 20,
        ScoreThreshold = threshold,
        WithPayload = true,
        Filter = new Filter
        {
            Must = new[]
            {
                new FieldCondition
                {
                    Key = "timestamp",
                    Range = new RangeCondition
                    {
                        Gte = DateTime.UtcNow.AddDays(-30)
                    }
                }
            }
        }
    });
    
    return searchResult.Select(ConvertToMemory).ToList();
}

  • HNSW index for vector similarity search
  • Payload indexes on conversation_id, session_id, timestamp
  • Batch processing for bulk operations
  • Connection pooling for Qdrant client

Security & Privacy

Critical Security Measures:
  • All memories encrypted at rest in Qdrant
  • TLS encryption for all network communication
  • API key rotation every 90 days
  • Audit logging for all memory operations
  • PII detection and masking before storage
